#include <stdio.h>

#include <stdlib.h>

 

#define STACK_SIZE 100

 

int stack[STACK_SIZE];

int top = -1;

 

// Push value (allocate memory)

void push(int value) {

   if (top == STACK_SIZE - 1) {

       printf("Stack Overflow! Cannot allocate more memory.\n");

       return;

   }

   stack[++top] = value;

   printf("Allocated variable with value: %d (at address %d)\n", value, top);

}

 

// Pop value (deallocate memory)

void pop() {

   if (top == -1) {

       printf("Stack Underflow! No memory to deallocate.\n");

       return;

   }

   printf("Deallocated variable with value: %d (from address %d)\n", stack[top], top);

   top--;

}

 

// Display stack content

void display() {

   if (top == -1) {

       printf("Stack is empty.\n");

       return;

   }

   printf("Current Stack (top to bottom):\n");

   for (int i = top; i >= 0; i--)

       printf("| %d |\n", stack[i]);

   printf("-----\n");

}

 

int main() {

   printf("=== Stack Allocation Simulation ===\n");

   push(10);

   push(20);

   push(30);

   display();

 

   pop();

   display();

 

   return 0;

}
